LintCode 逆波兰表达式求值

13/Oct 2016

题目链接

题意很明确，其实就是四则运算的后缀求值，用栈来维护，中间过程中没有考虑栈元素不够或者除数为0的情况，但也AC了，说明输入的数据比较严格。

int evalRPN(vector<string>& tokens) {
    // Write your code here
    stack<int> s;
    int ans;
    if (tokens.empty()) return ans;
    for (auto e : tokens) {
        if (e != "+" && e != "-" && e != "*" && e != "/") {
            int number = stoi(e);
            s.push(number);
            continue;
        } 
        int x = s.top();
        s.pop();
        int y = s.top();
        s.pop();
        if (e == "+") s.push(x + y);
        if (e == "-") s.push(y - x);
        if (e == "/") s.push(y / x);
        if (e == "*") s.push(x * y);
    }

    return s.top();
}
