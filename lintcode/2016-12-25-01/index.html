lintcode 矩阵的之字型遍历

25/Dec 2016

题目要求：给一个矩阵，按照之字型进行遍历，题目链接 思路：其实仔细想一想，就是先斜着往下遍历，再斜着往上遍历，就是这个过程，主要是判断好临界条件，是要往右走还是往下走 代码：

class Solution {
public:
    /**
     * @param matrix: a matrix of integers
     * @return: a vector of integers
     */
    vector<int> printZMatrix(vector<vector<int> > &matrix) {
        // write your code here
        int row = matrix.size();
        int col = matrix[0].size();
        vector<int> ret (row * col, 0);
        int r = 0;
        int c = 0;
        int i = 0;
        int all = row * col;
        ret[i++] = matrix[0][0];
        while (i < all) {
            // 斜上走到顶
            while (i < all && r - 1 >= 0 && c + 1 < col) {
                ret[i++] = matrix[--r][++c];
            }
            // 横右走一步，不可横右走时竖下走一步
            if (i < all && c + 1 < col) {
                ret[i++] = matrix[r][++c];
            } else if (i < all && r + 1 < row) {
                ret[i++] = matrix[++r][c];
            }
            // 斜下走到底
            while (i < all && r + 1 < row && c - 1 >= 0) {
                ret[i++] = matrix[++r][--c];
            }
            // 竖下走一步，不可竖下时横右走一步
            if (i < all && r + 1 < row) {
                ret[i++] = matrix[++r][c];
            } else if (i < all && c + 1 < col) {
                ret[i++] = matrix[r][++c];
            }
        }
        return ret;
    }
};
