LintCode 二进制表示

13/Oct 2016

题目链接

说是困难题，其实很简单的，就是将字符串中整数部分和小数部分分别提取出来，然后根据整数化二进制和小数划二进制的形式来求。唯一有问题的就是字符串中的小数部分化成小数时，如果按照传统的一个个来做的话，在计算机中表示很奇怪，例如对于4096.6435546875，小数部分按照6*10（-1） + 4*10（-2）…这样求的话，在计算机中实际表示为0.64355468750000011，所以很奇怪，估计是和计算机中浮点型的存储方式有关，所以后来就直接用C++自带的库函数atof()函数来转换，这样就ok了。

class Solution {
public:
/**
 *@param n: Given a decimal number that is passed in as a string
 *@return: A string
 */
string binaryRepresentation(string n) {
    // wirte your code here
    string ret;
    if (n.empty()) return 0;
    int len = n.size();
    int i = 0;
    while (i < len && n[i] != '.') ++i;
    int j = 0;
    bool flag = true;
    if (n[0] == '-') {
        flag = false;
        ++j;
        ret += "1";
    }

    long number = 0;
    while (j < i) {
        number = number * 10 + (n[j] - '0');
        ++j;
    }
    longToBinary(number, ret);

    if (i == len) return ret;
    ret += ".";


    double number1 = 0.0;
    ++i;
    int count = -1;
    while (i < len) {
        number1 += (n[i] - '0') * (double)pow(10, count);
        ++i;
        --count;
    }

    bool yes = true;
    // 自己计算的不行，还得靠库函数
    double decPart = atof(n.substr(n.find("."), n.size() - n.find(".")).c_str());
    number1 = decPart;
    yes = digitToBinary(number1, ret);

    int len1 = ret.size();
    if (ret[len1-1] == '.') ret = ret.substr(0,len1-1);
    if (yes) return ret;
    ret = "ERROR";
    return ret;
}

bool digitToBinary(double number, string &s) {
    int time = 0;
    double number1 = 1.0;
    double number2 = 0.0;
    while (time <= 32) {
        if (number == number1) 
            break;
        if (number == number2) 
            break;
        number *= 2;
        if (number >= 1) {
            number -= 1.0;
            s += to_string(1);
        } else s += to_string(0);

        ++time;
    }

    if (time <= 32) return true;
    return false;
}
void longToBinary(long number, string &s) {
    vector<int> vb;
    while (number != 0) {
        vb.push_back(number % 2) ;
        number /= 2;
    }
    if (vb.empty()) {
        s += to_string(0);
        return;
    }
    reverse(vb.begin(), vb.end());
    for (auto e : vb) {
        s += to_string(e);
    }

    return ;
}

};
