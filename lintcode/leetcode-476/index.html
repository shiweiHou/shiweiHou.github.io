leetcode 476.Number Complement

24/Feb 2017

class Solution { public: int findComplement(int num) { // 这个是最笨的一种方法，就是首先将num右移，查看num去除前缀0之后有效位有几位 // 然后再重新遍历一边num的末尾数字，根据有效位的个数控制右移几次，还利用了pow函数 // 网上有一种最简单的，其实上面的那种方法，也是根据后几位来求，假如后4位为有效位 // 那么我们如果能得到一个 后四位为0前面几位都为1的数，和num进行分别取反进行&操作就可以 // mask = ~0; // while (mask & num) mask = mask << 1; // return ~num & ~mask // 例如 num = 0000 0101 // 经过while循环， mask = 1111 1000 // ~num & ~mask = 0000 0010 = 2 int n = 0; int num1 = num; while (num1) { ++n; num1 = num1 >> 1; } int ret = 0, i = 0; while(i < n) { int a = num & 1; ret += (a == 1? 0 : 1) * pow(2,i); ++i; num = num >> 1; } return ret; } };
