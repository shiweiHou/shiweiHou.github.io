网易2017秋招编程题4

15/Feb 2017

数字翻转：

#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int rev(int num){
    vector<int> digit; 
    int t = 0; // 记录位数
    while (num) {
        digit.push_back(num % 10);
        num /= 10;
        ++t;
    }
    num = 0;
    for (int i = 0; i < t ; ++i) {
        num = num * 10 + digit[i];
    }
    return num;
}

int main() {
    int x, y;
    cin >> x >> y;
    cout << rev(rev(x) + rev(y)) << endl;
    return 0;
}
