网易2017秋招编程题1

15/Feb 2017

牛课网上做的： 回文序列：

#include <iostream>
#include <deque>
using namespace std;

bool is_hui_wen(deque<int> &dq) { // 判断是不是回文序列
    if (dq.empty()) return true;
    deque<int>::iterator beg = dq.begin();
    deque<int>::iterator en  = dq.end();
    --en;
    while ( beg < en) {
        if (*beg != *en) return false;
        ++beg;
        --en;
    }
    return true;
}

int min_change(deque<int> &dq) {
    int t = 0;
    while (!is_hui_wen(dq)) {// 如果不是，继续进行处理
        int b1 = *(dq.begin());
        dq.pop_front();
        int c1 = *(--dq.end());
        dq.pop_back();
        if (b1 == c1) continue; // 相等说明这个序列前后数字相同，略过，查看剩下的序列
        if (dq.empty() == false) { // 如果序列中还有其它元素
            if (b1 > c1) { // 前面的大，说明后面的需要两个数相加
                dq.push_front(b1);
                int c2 = *(--dq.end());
                c1 += c2;
                dq.pop_back();
                dq.push_back(c1);
            } else if (b1 < c1) { // 否则，前面的两个数相加
                dq.push_back(c1);
                int b2 = *(dq.begin());
                b1 += b2;
                dq.pop_front();
                dq.push_front(b1);
            }
        }
        ++t;
    }
    return t;
}

int main() {
    int n;
    cin >> n;
    deque<int> dq; // 利用双向队列来做，比较方便
    for (int i = 0; i < n; ++i) {
        int num;
        cin >> num;
        dq.push_back(num);
    }
    cout << min_change(dq) << endl;
    return 0;
}
