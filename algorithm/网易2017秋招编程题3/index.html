网易2017秋招编程题3

15/Feb 2017

跳石板：

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void find_yue_shu(int num, vector<int> &yue_shu) {
    for (int i = 2; i <= sqrt(num); ++i) {
        if (num % i == 0) {
            yue_shu.push_back(i);
            if (num / i != i) yue_shu.push_back(num / i);
        }
    }

}

int fuc(int N, int M) {
    if (N == M) return 0;
    vector<int> dp(M+1, -1);
    dp[N] = 0;
    for (int i = N; i <= M; ++i) {
        if (dp[i] == -1) continue; // 说明该点无法到达
        vector<int> yue_shu;
        find_yue_shu(i, yue_shu);
        for (int j = 0; j < yue_shu.size(); ++j) {
            int pos = yue_shu[j] + i; // 根据这个约数能到达的台阶位置
            if (pos <= M && dp[pos] != -1) { // 这个位置小于M，并且该位置已经可以到达了，就要进行比较,否则直接+1
                dp[pos] = min(dp[pos] , dp[i] + 1);
            } else if (pos <= M) {
                dp[pos] = dp[i] + 1;
            }
        }
    }

    if (dp[M] == -1) return -1;
    return dp[M];
}

int main() {
    int N, M;
    cin >> N >> M;
    cout << fuc(N, M) << endl;
    return 0;
}
