网易2017秋招编程题7

15/Feb 2017

计算糖果：

#include <iostream>
using namespace std;

int main() {
    int a, b, c, d;
    int A, B, C;
    cin >> a >> b >> c >> d;
    A = (a + c) / 2;
    B = A - a;
    C = B - b;
    if (A >= 0 && B >= 0 && C >=0 && (A-B) == a && (B-C) == b && (A+B) == c && (B+C) == d) cout << A << " " << B << " " << C << endl;
    else cout << "No" << endl;
    return 0;
}
